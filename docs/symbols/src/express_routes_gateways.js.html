1 var express = require('express'),
  2     router = express.Router(),
  3     _ = require('lodash'),
  4     logger = require('log4js').getLogger('HTTP');
  5 
  6 var utils = require('../utils'),
  7     db = require('../../lib/db');

  9 /**
 10  * retrieve item from database
 11  * 
 12  * @param {string} dbType
 13  * @returns http response
 14  */
 15 function getItem(dbType) {
 16   return function (req, res) {
 17     var ownerId = req.params.owner,
 18         id = req.params.id,
 19         ownerItem, resultItem;
 20 
 21     if (ownerId) {
 22       ownerItem = db.findOne('gateways', ownerId);
 23 
 24       if (!ownerItem) {
 25         logger.warn('[getItem] gateways:%s not found', ownerId);
 26         return res.status(404).send({error: 'gateways:' + ownerId +  ' is not found'});        
 27       }
 28     }
 29 
 30     resultItem = db.findOne(dbType, id);
 31 
 32     if (_.isUndefined(resultItem)) {
 33       logger.warn('[getitem] %s:%s not found', dbType, id);
 34       return res.status(404).send({error: dbType +  ':' + id + ' is not found'});
 35     } else {
 36 
 37       if (ownerId && ownerId != resultItem.owner) {
 38         return res.status(404).send({
 39           error: dbType + ':' + id + ' not match owner. ownerId= ' + ownerId + ', item owner=' + resultItem.owner
 40         });
 41       }
 42 
 43       return res.status(200).send(utils.processQuery(resultItem, req.query));
 44     }  
 45 
 46   };
 47 }

 49 /**
 50  * retrieve collection from database
 51  * 
 52  * @param {string} dbType
 53  * @returns
 54  */
 55 function getCollection(dbType) {
 56   return function (req, res) {
 57     var ownerId = req.params.owner,
 58         ownerItem, items, resultItems;
 59 
 60     if (ownerId) {
 61       ownerItem = db.findOne('gateways', ownerId);
 62 
 63       if (!ownerItem) {
 64         logger.warn('[getCollection] gateways:%s not found', ownerId);
 65         return res.status(404).send({error: 'gateways:' + ownerId +  ' is not found'});        
 66       }
 67     }
 68 
 69     items = db.findAll(dbType);
 70 
 71 
 72     if (_.isUndefined(items)) {
 73       logger.warn('[getCollection] %s not found', dbType);
 74       return res.status(404).send({error: dbType + ' is not found'});
 75     } else {
 76 
 77       if (ownerId) {
 78         resultItems = _.filter(items, { owner : ownerId});  
 79       } else {
 80         resultItems = items;
 81       }
 82 
 83       return res.status(200).send(utils.processQuery(resultItems, req.query));
 84     }  
 85   };
 86 }

 88 /**
 89  * check validation and insert into database
 90  * 
 91  * @param {string} dbType
 92  * @returns
 93  */
 94 function postItem(dbType) {
 95   return function (req, res) {
 96     var ownerId = req.params.owner,
 97         item = req.body,
 98         tcRest = req.app.locals.tcRest,
 99         ownerItem, tcFunc;
100 
101     if (ownerId) {
102       ownerItem = db.findOne('gateways', ownerId);
103 
104       if (!ownerItem) {
105         logger.warn('[getCollection] gateways:%s not found', ownerId);
106         return res.status(404).send({error: 'gateways:' + ownerId +  ' is not found'});        
107       }
108     }
109 
110     if (dbType === 'devices') {
111       tcFunc = tcRest.devicePost;
112     } else if (dbType === 'sensors') {
113       tcFunc = tcRest.sensorPost;
114     }
115 
116     tcFunc(tcRest, ownerItem, item, Date.now(), function (err, result) {
117       if (err) {
118         return res.status(400).send({error: err}); 
119       }
120 
121       // rename reqId to id
122       result.id = result.reqId;
123       delete result.reqid;
124 
125       // result id insert into 'gateways'
126       ownerItem[dbType].push(result.id);
127       db.updateOne('gateways', ownerId, ownerItem);
128 
129       // insert into dbTypes(devices or sensors)
130       // set owner 
131       // set ctime if ctime is empty
132       result.owner = ownerId;
133       if (!result.ctime) {
134         result.ctime = Date.now().toString();
135       }
136       db.insertOne(dbType, result);
137 
138       return res.status(201).send(result);
139     });
140 
141   };
142 }
