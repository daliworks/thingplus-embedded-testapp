 37     config = JSON.parse(file);

 42     logger.error('[loadConfig] config json pasing failed. error=', e);

 60 /**
 61  * generate id with token
 62  *
 63  * @param {string} str
 64  * @param {object} tokens
 65  * @returns
 66  */
 67 function registerIdTemplate(str, tokens) {
 68  return str && str.replace(/\{(\w+)\}/g, function (x, key) {
 69     return tokens[key];
 70   });
 71 }

 73 /**
 74  * create sensor data with sensorDrive info
 75  *
 76  * @param {string} gatewayId
 77  * @param {string} deviceId
 78  * @param {object} sensorModel
 79  * @returns
 80  */
 81 function registerSensor(gatewayId, deviceId, sensorModel) {
 82   var sensor, sensorId, sensorDriverInfo;
 83 
 84   sensorDriverInfo = database.findOne('sensorDrivers', sensorModel.driverName);
 85 
 86   sensorId = registerIdTemplate(sensorDriverInfo.idTemplate, {
 87                     model: sensorModel.model,
 88                     gatewayId: gatewayId,
 89                     // FIXME: where is deviceAddress in GatewayModel???
 90                     deviceAddress: deviceId,
 91                     sequence: sensorModel.sequence,
 92                     type: sensorModel.type
 93                   });
 94 
 95   sensor = _.clone(sensorModel);
 96   sensor.id = sensorId;
 97   sensor.owner = gatewayId;
 98   sensor.deviceId = deviceId;
 99   sensor.name = sensor.type + '_' + (_.isUndefined(sensor.sequence) ? 1 : sensor.sequence);
100   sensor.ctime = (Date.now()).toString();
101 
102   return sensor;
103 }

105 /**
106  * create device data with device model info
107  *
108  * @param {string} gatewayId
109  * @param {object} deviceModel
110  * @returns
111  */
112 function registerDevice(gatewayId, deviceModel) {
113 
114   var device, deviceId;
115 
116   deviceId = registerIdTemplate(deviceModel.idTemplate, {
117     gatewayId: gatewayId,
118     // FIXME: where is deviceAddress in GatewayModel???
119     deviceAddress: deviceModel.id
120   });
121 
122   device = {
123     id: deviceId || gatewayId,
124     model: deviceModel.id,
125     name: deviceModel.displayName,
126     owner: gatewayId,
127     ctime: (Date.now()).toString()
128   };
129 
130   return device;
131 }

133 /**
134  * create gateway data and insert into database
135  *
136  * @param {object} config
137  * @returns
138  */
139 function registerGateway(config) {
140   var gateway = {}, devices = [], sensors = [], gatewayOpt = config.gateway, gatewayModelInfo;
141 
142   if (!gatewayOpt) {
143     logger.error('[registerGateway] undefined config.gateway');
144     return;
145   }
146 
147   if (_.isUndefined(gatewayOpt.id)) {
148     logger.error('[registerGateway] not found id. config=', gatewayOpt);
149     return;
150   }
151 
152   gateway.id = gatewayOpt.id;
153 
154   if (_.isUndefined(gatewayOpt.apikey)) {
155     logger.error('[registerGateway] not found apikey. config=', gatewayOpt);
156     return;
157   }
158 
159   gateway.apikey = gatewayOpt.apikey;
160 
161   if (_.isUndefined(gatewayOpt.reportInterval) || isNaN(gatewayOpt.reportInterval)) {
162     logger.error('[registerGateway] not found reportInterval or not a number. config=', gatewayOpt);
163     return;
164   }
165 
166   // second to millisecond and change to string
167   gateway.reportInterval = (parseInt(gatewayOpt.reportInterval, 10) * 1000).toString();
168 
169   if (_.isUndefined(gatewayOpt.model)) {
170     logger.error('[registerGateway] not found model. config=', gatewayOpt);
171     return;
172   }
173 
174   gateway.model = gatewayOpt.model;
175   gatewayModelInfo = database.findOne('gatewayModels', gateway.model);
176 
177   if (!gatewayModelInfo) {
178     logger.error('[registerGateway] invalid gateway model. model id=', gateway.model);
179     return;
180   }
181 
182   gateway.name = gatewayModelInfo.displayName;
183 
184   // register devices
185   _.forEach(gatewayModelInfo.deviceModels, function (deviceModel) {
186     var registerDeviceInfo;
187 
188     if (deviceModel.discoverable === 'n') {
189       registerDeviceInfo = registerDevice(gateway.id, deviceModel);
190       _.forEach(deviceModel.sensors, function (sensorModel) {
191         var registerSensorInfo = registerSensor(gateway.id, deviceModel.id, sensorModel);
192 
193         if (registerSensorInfo) {
194           sensors.push(registerSensorInfo);
195         }
196       });
197     }
198 
199     if (registerDeviceInfo) {
200       devices.push(registerDeviceInfo);
201     }
202   });
203 
204   gateway.devices = _.map(devices, 'id');
205 
206   gateway.sensors = _.map(sensors, 'id');
207   gateway.ctime = (Date.now()).toString();
208 
209   database.insertOne('gateways', gateway);
210   database.insertMany('sensors', sensors);
211   database.insertMany('devices', devices);
212 }

214 /**
215  * start mqtt broker(mosca)
216  *
217  * @param {object} config
218  * @param {object} tc
219  * @param {function} cb
220  * @returns
221  */
222 function runMqttBroker(config, tc, cb) {
223   mqttBroker = mosca.start(config);
224 
225   if (!mqttBroker) {
226     logger.error('[runMqttBroker] create failed');
227     return cb && cb('mqttBroker create failed');
228   }
229 
230   mqttBroker.on('clientConnected', function (client) {
231     var time = new Date();
232 
233     logger.info('[runMqttBroker] MQTT CONNECTED. clinet', client);
234 
235     tc.mqtt.testcases.cleanSession(client.clean, time);
236     tc.mqtt.testcases.keepalive(client.keepalive, time);
237     tc.mqtt.testcases.willMessage(client.will, time);
238 
239     tc.mqtt.on('publishMqttMessage', function (message) {
240       mqttBroker.publish(message);
241       //console.log(message);
242     });
243 
244     setTimeout(tc.mqtt.sendActuatorCommands(tc.mqtt), 1000);
245   });
246 
247   mqttBroker.on('published', function (packet, client) {
248     var time = new Date();
249 
250     if (_.startsWith(packet.topic, '$SYS')) {
251       return;
252     }
253 
254     if (_.endsWith(packet.topic, 'req')) {
255       return;
256     }
257 
258     tc.mqttParser.parse(packet.topic, packet.payload.toString(), time, function (err, msg) {
259       if (err) {
260         logger.error('[runMqttBroker/published] mqtt parsing failed. error=', err);
261         return;
262       }
263 
264       if (msg.status) {
265         //tc.mqtt.statuses(msg.status);
266         tc.mqtt.testcases.status(msg.status, time);
267       }
268 
269       if (msg.sensorValue) {
270         //tc.mqtt.sensorValues(msg.sensorValue);
271         tc.mqtt.testcases.sensorValues(msg.sensorValue);
272       }
273 
274       if (msg.response) {
275         //testcases.mqtt.tcActuator(msg.response);
276         tc.mqtt.testcases.actuatorResponse(msg.response);
277       }
278     });
279   });
280 
281   mqttBroker.on('gatewayId', function (gatewayId) {
282     tc.mqtt.testcases.gatewayId(gatewayId);
283   });
284 
285   mqttBroker.on('apikey', function (apikey) {
286     //tc.mqtt.tcApikey(apikey.toString());
287     tc.mqtt.testcases.apikey(apikey.toString());
288   });
289 
290   mqttBroker.on('keepalive', function (keepalive) {
291     //tc.mqtt.tcKeepalive(keepalive);
292     tc.mqtt.testcases.keepalive(keepalive);
293   });
294 
295   return cb && cb();
296 }

298 /**
299  * start http server(express)
300  *
301  * @param {object} config
302  * @param {object} tc
303  * @param {function} cb
304  * @returns
305  */
306 function runHttpServer(config, tc, cb) {
307   var port = config.http && config.http.port || 3000;
308 
309   expressApp.set('port', port);
310 
311   expressApp.locals.tcRest = tc.rest;
312 
313   httpServer = http.createServer(expressApp);
314 
315   httpServer.listen(port);
316 
317   httpServer.on('error', function (error) {
318     if (error.syscall !== 'listen') {
319       throw error;
320     }
321 
322     var bind = typeof port === 'string' ? 'Pipe ' + port : 'Port ' + port;
323 
324     // handle specific listen errors with friendly messages
325     switch (error.code) {
326       case 'EACCES':
327         logger.error('[runHttpServer/error] ' + bind + ' requires elevated privileges');
328         process.exit(1);
329         break;
330       case 'EADDRINUSE':
331         logger.error('[runHttpServer/error] ' + bind + ' is already in use');
332         process.exit(1);
333         break;
334       default:
335         throw error;
336     }
337   });
338 
339   httpServer.on('listening', function onListening() {
340     var addr = httpServer.address();
341     var bind = typeof addr === 'string' ? 'pipe ' + addr : 'port ' + addr.port;
342     logger.info('[runHttpServer] Listening on ' + bind);
343   });
344 
345   return cb && cb();
346 }

357 /**
358  * start local-test app
359  *
360  * @returns
361  */
362 function start() {
363   var config = loadConfig();
364 
365   if (!config) {
366     logger.error('[start] loading config failed');
367     return;
368   }
369 
370   database.init();
371 
372   testcases = createTestCase(config);
373 
374   registerGateway(config);
375 
376   if (!testcases) {
377     logger.error('[start] create test case failed');
378     return;
379   }
380 
381   runMqttBroker(config, testcases, function (err) {
382     if (err) {
383       logger.error('[start] run mqtt broker failed. error=', err);
384     } else {
385       logger.info('[start] run mqtt broker');
386     }
387   });
388 
389   runHttpServer(config, testcases, function (err) {
390     if (err) {
391       logger.error('[start] run http server failed. error=', err);
392     } else {
393       logger.info('[start] run http server');
394     }
395   });
396 }
